HPAL,L,"TEN"
BEGIN
   COMMENT THIS PROGRAM PRINTS THE SQUARE ROOTS
           OF THE FIRST TEN INTEGERS
           ALGOL 60 FOR HP2100;
   INTEGER I;
   REAL R;
   FORMAT F1(I3, 3X, F10.5);
   WRITE(1, #("HELLO FROM ALGOL!  HERE ARE SOME SQUARE ROOTS:"/));
   FOR I := 1 STEP 1 UNTIL 10 DO
     BEGIN
       R := SQRT (I);
       WRITE (1, F1, I, R);
     END;
END$
