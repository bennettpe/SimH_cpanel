      PROGRAM GMF2
C 
      LU1350=18
C.....   SET THE ASCII EXT (END OF CODE, ASCII CHAR 03)
C.....   IN THE UPPER 8 BITS
      IETX=3*256
C.....   SET THE ASCII DC4 IN THE UPPER 8 BITS
      IDC4=20*256
C.....   INITIALIZE THE HP1350A
      WRITE(LU1350,99) IETX,IDC4
   99 FORMAT(A1,A1,":SN:SX:EM:EN:EX:UM:")
      DO 199 I=0,3
      CS=I
      IF (I.EQ.0) GO TO 100
      IF (I.EQ.1) GO TO 101
      IF (I.EQ.2) GO TO 102
      IF (I.EQ.3) GO TO 103
  100 X=0
      Y=1000
      GO TO 200
  101 X=0
      Y=900
      GO TO 200
  102 X=0
      Y=700
      GO TO 200
  103 X=0
      Y=350
      GO TO 200
C.....   DISPLAY CHARSET AT X,Y WITH SIZE CS 
C.....   MOVE THE BEAM TO THE STARTING POSITION
  200 WRITE(1,1) CS,X,Y
    1 FORMAT("CS=",I1," X=",I4," Y=", I4)
      WRITE(LU1350,2) X,Y
    2 FORMAT("PE0,:PA",I4,",",I4,";:PE1,:")
C.....   SETUP CHAR PRINTING
      WRITE(LU1350,3) CS
    3 FORMAT("CS",I1,",:")
C.....   PRINT CHARS 
      WRITE(LU1350,4)
    4 FORMAT("TXABCDEFGHIJKLMNOPQRTSUVWXYZ")
      WRITE(LU1350,5)
    5 FORMAT("!#$%&()*+,.;:_-/<=>?@^")
      WRITE(LU1350,6)
    6 FORMAT("0123456789")
      C0=34*256
      C1=39*256
      C2=92*256
      C3=123*256
      C4=124*256
      C5=125*256
      C6=126*256
      C7=127*256
      WRITE(LU1350,7) C0,C1,C2,C3,C4,C5,C6,C7
    7 FORMAT(A1,A1,A1,A1,A1,A1,A1,A1)
      WRITE(LU1350,8) IETX
    8 FORMAT(A1)
  199 CONTINUE
      STOP
      END
      END$
